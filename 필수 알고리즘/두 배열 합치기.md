## 두 배열 합치기

```java
import java.util.*;

public class Main {
    public List<Integer> solution(int n, int m, int[] aList, int[] bList) {
        List<Integer> answer = new ArrayList<>();
        int p1 = 0;
        int p2 = 0;

        while (p1 < n && p2 < m) {
            if (aList[p1] < bList[p2]) {
                answer.add(aList[p1++]);
            } else {
                answer.add(bList[p2++]);
            }
        }

        while (p1 < n) {
            answer.add(aList[p1++]);
        }

        while (p2 < m) {
            answer.add(bList[p2++]);
        }
        return answer;
    }

    public static void main(String[] args) {
        Main main = new Main();
        Scanner sc = new Scanner(System.in);

        int a = sc.nextInt();
        int[] aList = new int[a];
        for (int i = 0; i < a; i++) {
            aList[i] = sc.nextInt();
        }

        int b = sc.nextInt();
        int[] bList = new int[b];
        for (int i = 0; i < b; i++) {
            bList[i] = sc.nextInt();
        }

        for (int x : main.solution(a, b, aList, bList)) {
            System.out.print(x + " ");
        }
    }
}
```
