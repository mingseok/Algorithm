## 소수(에라토스테네스 체)

```java
package org.example;

import java.util.*;

public class Main {
    public static void main(String[] args) {
        Main main = new Main();
        Scanner sc = new Scanner(System.in);
        
        int num = sc.nextInt();
        System.out.println(main.solution(num));
    }

    public int solution(int n) {
        int answer = 0;
        int[] check = new int[n+1];

        for (int i = 2; i <= n; i++) {
            if (check[i] == 0) {
                answer++;
                for (int j = i; j <= n; j = j + i) { //2의 배수 모두
                    check[j] = 1;
                }
            }
        }
        return answer;
    }
}
```
