## 최대 매출

```java
package org.example;

import java.util.*;

class Main {
    public int solution(int a, int b, int[] arr) {
        int answer = 0;
        int index = 0;

        int left = 1;
        int right = 1;
        int max = 0;
        int count = 1;

        for (int i = 0; i < a; i++) {
            if (max < arr[i]) {
                max = arr[i];
                index = i;
            }
        }

        answer = max;
        while (count <= 2) {
            if (arr[index - left] < arr[index + right]) {
                answer += arr[index + right];

                right++;
                count++;
            } else if (arr[index - left] == arr[index + right]) {
                answer += arr[index - left];
                answer += arr[index - right];

                right++;
                left++;

                count++;
                count++;
            } else {
                answer += arr[index - left];

                left++;
                count++;
            }
        }

        return answer;
    }

    public static void main(String[] args) {
        Main main = new Main();
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int num2 = sc.nextInt();
        int[] arr = new int[num];

        for (int i = 0; i < num; i++) {
            arr[i] = sc.nextInt();
        }

        System.out.println(main.solution(num, num2, arr));
    }
}
```
