## 임시반장정하기

```java
import java.util.Scanner;

public class Main {

    public int solution(int num, int[][] arr) {
        int max = 0;
        int index = 0;

        for (int i = 0; i < num; i++) {
            int count = 0;

            for (int j = 0; j < num; j++) {
                if (i == j) {
                    continue;
                }

                if (arr[i][0] == arr[j][0] ||
                    arr[i][1] == arr[j][1] ||
                    arr[i][2] == arr[j][2] ||
                    arr[i][3] == arr[j][3] ||
                    arr[i][4] == arr[j][4]) {
                    count++;
                }
            }

            if (max < count) {
                max = count;
                index = i + 1;
            }
        }
        return index;
    }

    public static void main(String[] args) {
        Main main = new Main();
        Scanner sc = new Scanner(System.in);

        int num = sc.nextInt();
        int[][] arr = new int[num][5];

        for (int i = 0; i < num; i++) {
            for (int j = 0; j < 5; j++) {
                arr[i][j] = sc.nextInt();
            }
        }
        System.out.println(main.solution(num, arr));
    }
}
```
